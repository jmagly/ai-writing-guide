### Task 4 – MCP server works from any subdirectory (30 min)

**Title:** Auto-find project root in MCP server

**File:** `src/mcp/server.mjs`

```javascript
import { findUp } from "find-up";
import path from "path";

async function findProjectRoot() {
  const aiwgDir = await findUp(".aiwg", { type: "directory" });
  if (!aiwgDir) {
    throw new Error("No .aiwg directory found. Run from an AIWG project or `aiwg -new` first.");
  }
  return path.dirname(aiwgDir);
}
```

Use `await findProjectRoot()` instead of `process.cwd()` everywhere.

### Task 5 – Friendly PATH warning after npm install (20 min)

**Title:** postinstall script with colorful warning

**Add to package.json:**
```json
"scripts": {
  "postinstall": "node ./bin/postinstall.mjs"
}
```

**Create `bin/postinstall.mjs`:**
```javascript
#!/usr/bin/env node
import { execSync } from "child_process";

try {
  execSync("aiwg --version", { stdio: "ignore" });
} catch {
  console.log("\n⚠️  aiwg installed but not in PATH!");
  console.log("   Run this to fix it:\n");
  console.log("   npm config set prefix ~/npm-global");
  console.log("   export PATH=~/npm-global/bin:$PATH   # add to ~/.zshrc or ~/.bashrc\n");
}
```
